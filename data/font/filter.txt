NameLibrary::NameLibrary(){
	int fnNum = sizeof(g_familyNameData)/sizeof(g_familyNameData[0]);
	float autoWeight = 98;
	float weightStep = 0.4f;
	_totalFamilyNameWeight = 0;
	for ( int i = 0; i < fnNum; ++i ){
		NameData& d = g_familyNameData[i];
		if ( d.weight == 0){
			d.weight = (int)autoWeight;
			autoWeight -= weightStep;
			autoWeight = max(1.1f, autoWeight);
		}
		_totalFamilyNameWeight += d.weight;
		_familyNameMap[_totalFamilyNameWeight] = d.name;
	}

	fnNum = sizeof(g_maleSingleNameData)/sizeof(g_maleSingleNameData[0]);
	_totalMaleSingleNameWeight = 0;
	for ( int i = 0; i < fnNum; ++i ){
		NameData& d = g_maleSingleNameData[i];
		_totalMaleSingleNameWeight += (int)(d.weight*0.01f);
		_maleSingleNameMap[_totalMaleSingleNameWeight] = d.name;
	}

	fnNum = sizeof(g_femaleSingleNameData)/sizeof(g_femaleSingleNameData[0]);
	_totalFemaleSingleNameWeight = 0;
	for ( int i = 0; i < fnNum; ++i ){
		NameData& d = g_femaleSingleNameData[i];
		_totalFemaleSingleNameWeight += (int)(d.weight*0.01f);
		_femaleSingleNameMap[_totalFemaleSingleNameWeight] = d.name;
	}

	fnNum = sizeof(g_secondNameData)/sizeof(g_secondNameData[0]);
	_totalSecondNameWeight = 0;
	for ( int i = 0; i < fnNum; ++i ){
		NameData& d = g_secondNameData[i];
		_totalSecondNameWeight += (int)(d.weight*0.01f);
		_secondNameMap[_totalSecondNameWeight] = d.name;
	}

	fnNum = sizeof(g_maleThirdNameData)/sizeof(g_maleThirdNameData[0]);
	_totalMaleThirdNameWeight = 0;
	for ( int i = 0; i < fnNum; ++i ){
		NameData& d = g_maleThirdNameData[i];
		_totalMaleThirdNameWeight += (int)(d.weight*0.01f);
		_maleThirdNameMap[_totalMaleThirdNameWeight] = d.name;
	}

	fnNum = sizeof(g_femaleThirdNameData)/sizeof(g_femaleThirdNameData[0]);
	_totalFemaleThirdNameWeight = 0;
	for ( int i = 0; i < fnNum; ++i ){

		_totalFemaleThirdNameWeight += (int)(d.weight*0.01f);
		_femaleThirdNameMap[_totalFemaleThirdNameWeight] = d.name;
